#include <iostream>
#include <windows.h>
using namespace std;
int main() {
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);
    int n;
    cout << "Введіть ціле позитивне число: ";
    cin >> n;
    (n >= 10 && n <= 99 && n % 2 == 0)
        ? cout << "Число є парним двозначним" << endl
        : cout << "Число не є парним двозначним" << endl;
    system("pause");
    return 0;
}
